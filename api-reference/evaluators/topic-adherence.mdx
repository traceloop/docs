## Request Body

<ParamField body="input" type="object" required>
  <Expandable title="input object">
<ParamField body="completion" type="string" required>

</ParamField>

</ParamField>

<ParamField body="reference_topics" type="string" required>

</ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="adherence_score" type="number">

</ResponseField>
